消息类：com.eg.intf3.msg.QueryStatus

查询设备的运行状态，并可以从返回结果中得到运行状态对象

//指定主控器唯一标识符和设备通讯码
String ctrlUUID = "14:06:30:00:04:AB";
int deviceCode = 100;
//构建门锁状态查询对象
QueryStatus msg = new QueryStatus(deviceCode, ctrlUUID);
//下发消息并设置监听器
console.dispatchTask(msg, new OneTaskStateChangeAdapter(){
@Override
public void onSuccess(Task task) {
//成功后，可以得到门锁状态信息对象
LockStatusInfo lockStatusInfo = task.getResult_LockStatus();
if(lockStatusInfo == null)
    return;
//门锁状态的处理可以参考“接收到设备状态”
}

@Override
public void onFail(Task task) {

}
});
