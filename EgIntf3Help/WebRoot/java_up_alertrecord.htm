public void onUploadAlertRecord(UploadAlertRecord record) {

/**
* 报警类型：
* UploadAlertRecord.ALERT_TYPE_DISRUPT: 门锁防拆报警
*
* UploadAlertRecord.ALERT_TYPE_FORCED_ENTER: 门锁强行闯入报警
*
* UploadAlertRecord.ALERT_TYPE_POWER_OFF: 控制器断电报警
*
* UploadAlertRecord.ALERT_TYPE_BACKUP_POWER_LOW: 控制器备用电池低电量报警
*
* UploadAlertRecord.ALERT_TYPE_MANUAL: 手动报警
*
* UploadAlertRecord.ALERT_TYPE_DOOR_LOCK_FAILED: 门未关好报警（虚掩）
*
* UploadAlertRecord.ALERT_TYPE_BATTERY_POWER_LOW: （门锁）电池低电量
*
* UploadAlertRecord.ALERT_TYPE_LONG_TIME_OPENED: 门长时间未关报警
*
* UploadAlertRecord.ALERT_TYPE_FORCE_PIN: 使用胁迫密码报警
*
* UploadAlertRecord.ALERT_TYPE_CAPTURE_SD_OVERFLOW: 拍照器存储空间满（无法自动回收）。
*
* 还有其他一些报警类型，暂时用不到，以后可以扩展
*/
record.getAlertType();
//报警发生的时间戳，与 UTC 1970年时间差，以秒为单位
record.getTimeStamp();
//主控器唯一编号
record.getCtrlUUID();
//门锁编号，一般就是100
record.getDeviceCode();
}
