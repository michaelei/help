record：开门记录对象，Java类：com.eg.intf3.msg.UploadOpenRecord

示例代码

public void onUploadOpenRecord(UploadOpenRecord record) {

/**
* 开门类型：
* UploadOpenRecord.OpenType_UnKnow: 未知的开门类型，=0，错误
*
* UploadOpenRecord.OpenType_Card: 刷卡开门
*
* UploadOpenRecord.OpenType_PIN: 密码开门
*
* UploadOpenRecord.OpenType_Finger: 指纹开门
*
* UploadOpenRecord.OpenType_Card_PIN: 刷卡+密码
*
* UploadOpenRecord.OpenType_Card_Finger: 刷卡+指纹
*
* UploadOpenRecord.OpenType_Card_PIN_Finger: 刷卡+密码+指纹
*
* UploadOpenRecord.OpenType_PIN_Finger: 密码+指纹
*
* UploadOpenRecord.OpenType_Key: 机械钥匙开门
*
* UploadOpenRecord.OpenType_ExitButton: 出门按钮开门
*
* UploadOpenRecord.OpenType_Handler: 把手开门
*
* UploadOpenRecord.OpenType_External_Signal: 外部信号（对讲机）开门
*
* UploadOpenRecord.OpenType_Remote_Control: 远程命令开门
*
*/
record.getOpenType();
//开门发生的时间戳，与 UTC 1970年时间差，以秒为单位
record.getTimeStamp();
//开门的卡号，long型
record.getCardNoL();
//主控器唯一编号
record.getCtrlUUID();
//门锁编号，一般就是100
record.getDeviceCode();
//本次开门是否成功。0表示成功，其它数值代表错误代码
record.getResultCode();
}
