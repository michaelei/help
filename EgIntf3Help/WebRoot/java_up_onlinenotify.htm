public void onOnlineNotify(OnlineNotify onlineNotify)

onlineNotify：设备上线提醒对象，Java类：com.eg.intf3.msg.OnlineNotify

示例代码

public void onOnlineNotify(OnlineNotify onlineNotify) {

//获得主控器唯一编号
onlineNotify.getCtrlUUID();
//获得门锁的序列号，一般情况下，一个主控器只连接一把门锁，因此该编号=100
onlineNotify.getDeviceCode();
//获得门锁的联机状态
if(onlineNotify.getWorkStatus() == OnlineNotify.Status_OK){
        //=OnlineNotify.Status_OK表示该门锁工作正常
}else if(onlineNotify.getWorkStatus() == OnlineNotify.Status_Offline){
        //=OnlineNotify.Status_Offline表示该门锁已经脱机了
}else{
        //其它情况，暂时不用考虑，以后可能会扩展
}
}
